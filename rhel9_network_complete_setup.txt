================================================================================
SECTION 1: PRE-CHECK - CURRENT STATE
================================================================================

# Check OS version
cat /etc/os-release

# Check interface name
ip a

# Check current routes
ip r

# Check NetworkManager connections
nmcli connection show

# Check device details
nmcli device show ens160

# Check DNS
cat /etc/resolv.conf

# Check NTP status
timedatectl status


================================================================================
SECTION 2: STATIC IP CONFIGURATION
================================================================================

--------------------------------------------------------------------------------
2.1 Set Static IP Address (Disable DHCP)
--------------------------------------------------------------------------------

nmcli connection modify ens160 \
  ipv4.method manual \
  ipv4.addresses 192.168.121.55/24 \
  ipv4.gateway 192.168.121.2 \
  ipv4.dns 192.168.121.2 \
  ipv4.dns-search localdomain

nmcli connection modify ens160 ipv4.method manual ipv4.addresses 192.168.121.55/24 ipv4.gateway 192.168.121.2 ipv4.dns 192.168.121.2 ipv4.dns-search localdomain

# Apply changes
nmcli connection up ens160

# Verify
ip a show ens160


--------------------------------------------------------------------------------
2.2 Add Extra IP Address (Secondary IP)
--------------------------------------------------------------------------------

nmcli connection modify ens160 +ipv4.addresses 192.168.121.100/24

# Apply
nmcli connection up ens160

# Verify
ip a show ens160


--------------------------------------------------------------------------------
2.3 Remove an IP Address
--------------------------------------------------------------------------------

nmcli connection modify ens160 -ipv4.addresses 192.168.121.100/24

# Apply
nmcli connection up ens160

# Verify
ip a show ens160


--------------------------------------------------------------------------------
2.4 Expected Output
--------------------------------------------------------------------------------

# ip a show ens160
2: ens160: <BROADCAST,MULTICAST,UP,LOWER_UP> mtu 1500 ...
    inet 192.168.121.55/24 brd 192.168.121.255 scope global noprefixroute ens160
       valid_lft forever preferred_lft forever


================================================================================
SECTION 3: GATEWAY CONFIGURATION
================================================================================

--------------------------------------------------------------------------------
3.1 Set Default Gateway
--------------------------------------------------------------------------------

nmcli connection modify ens160 ipv4.gateway 192.168.121.2

# Apply
nmcli connection up ens160

# Verify
ip r | grep default


--------------------------------------------------------------------------------
3.2 Change Gateway
--------------------------------------------------------------------------------

nmcli connection modify ens160 ipv4.gateway 192.168.121.1

# Apply
nmcli connection up ens160


--------------------------------------------------------------------------------
3.3 Remove Gateway
--------------------------------------------------------------------------------

nmcli connection modify ens160 ipv4.gateway ""

# Apply
nmcli connection up ens160


--------------------------------------------------------------------------------
3.4 Expected Output
--------------------------------------------------------------------------------

# ip r
default via 192.168.121.2 dev ens160 proto static metric 100


================================================================================
SECTION 4: STATIC ROUTES CONFIGURATION
================================================================================

--------------------------------------------------------------------------------
4.1 Add Single Static Route
--------------------------------------------------------------------------------

nmcli connection modify ens160 +ipv4.routes "10.10.8.0/21 192.168.121.2"

# Apply
nmcli connection up ens160


--------------------------------------------------------------------------------
4.2 Add Multiple Static Routes
--------------------------------------------------------------------------------

nmcli connection modify ens160 ipv4.routes \
  "10.10.8.0/21 192.168.121.2, 10.20.0.0/16 192.168.121.254"

# Apply
nmcli connection up ens160


--------------------------------------------------------------------------------
4.3 Remove a Static Route
--------------------------------------------------------------------------------

nmcli connection modify ens160 -ipv4.routes "10.20.0.0/16 192.168.121.254"

# Apply
nmcli connection up ens160


--------------------------------------------------------------------------------
4.4 View All Routes
--------------------------------------------------------------------------------

ip r


--------------------------------------------------------------------------------
4.5 Temporary Route (lost on reboot)
--------------------------------------------------------------------------------

ip route add 10.30.0.0/16 via 192.168.121.2 dev ens160

# Remove temporary route
ip route del 10.30.0.0/16 via 192.168.121.2 dev ens160


--------------------------------------------------------------------------------
4.6 Expected Output
--------------------------------------------------------------------------------

# ip r
default via 192.168.121.2 dev ens160 proto static metric 100
10.10.8.0/21 via 192.168.121.2 dev ens160 proto static metric 100
192.168.121.0/24 dev ens160 proto kernel scope link src 192.168.121.55 metric 100

SECTION 5: DNS CONFIGURATION
================================================================================

--------------------------------------------------------------------------------
5.1 Set Primary DNS
--------------------------------------------------------------------------------

5.3 Set DNS Search Domain
--------------------------------------------------------------------------------

--------------------------------------------------------------------------------
5.4 Remove DNS
--------------------------------------------------------------------------------

--------------------------------------------------------------------------------
5.5 Verify DNS
--------------------------------------------------------------------------------

# Check resolv.conf

# Check via nmcli
nmcli connection show ens160 | grep DNS

# Test DNS resolution
nslookup google.com
dig google.com


--------------------------------------------------------------------------------
5.6 Expected Output
--------------------------------------------------------------------------------

# cat /etc/resolv.conf

nameserver 8.8.8.8

SECTION 6: NTP CONFIGURATION
================================================================================

--------------------------------------------------------------------------------
6.1 Check Current NTP Status
--------------------------------------------------------------------------------

timedatectl status
timedatectl show-timesync


--------------------------------------------------------------------------------
6.2 Check NTP Service
--------------------------------------------------------------------------------

systemctl status chronyd


--------------------------------------------------------------------------------
6.3 View Current NTP Servers
--------------------------------------------------------------------------------

cat /etc/chrony.conf | grep server
chronyc sources -v


--------------------------------------------------------------------------------
6.4 Add Custom NTP Server
--------------------------------------------------------------------------------

# Edit chrony config
vi /etc/chrony.conf

# Add your NTP server line (example):
server 192.168.121.2 iburst
server 0.rhel.pool.ntp.org iburst
server 1.rhel.pool.ntp.org iburst


--------------------------------------------------------------------------------
6.5 Restart and Enable NTP Service
--------------------------------------------------------------------------------

systemctl restart chronyd
systemctl enable chronyd


--------------------------------------------------------------------------------
6.6 Set Timezone
--------------------------------------------------------------------------------

# List available timezones
timedatectl list-timezones | grep Asia

# Set timezone (example: India)
timedatectl set-timezone Asia/Kolkata

--------------------------------------------------------------------------------

================================================================================
SECTION 8: QUICK REFERENCE - ALL COMMANDS
================================================================================

STATIC IP
---------
Set static IP          : nmcli connection modify ens160 ipv4.method manual ipv4.addresses 192.168.121.55/24
Add extra IP           : nmcli connection modify ens160 +ipv4.addresses 192.168.121.100/24
Remove IP              : nmcli connection modify ens160 -ipv4.addresses 192.168.121.100/24

GATEWAY
-------
Set gateway            : nmcli connection modify ens160 ipv4.gateway 192.168.121.2
Remove gateway         : nmcli connection modify ens160 ipv4.gateway ""

ROUTES
------
Add route              : nmcli connection modify ens160 +ipv4.routes "10.10.8.0/21 192.168.121.2"
Remove route           : nmcli connection modify ens160 -ipv4.routes "10.10.8.0/21 192.168.121.2"
View routes            : ip r

DNS
---
Set DNS                : nmcli connection modify ens160 ipv4.dns "192.168.121.2 8.8.8.8"
Set search domain      : nmcli connection modify ens160 ipv4.dns-search localdomain
Remove DNS             : nmcli connection modify ens160 ipv4.dns ""
Verify DNS             : cat /etc/resolv.conf

NTP
---
Check NTP status       : timedatectl status
Enable NTP             : timedatectl set-ntp true
Set timezone           : timedatectl set-timezone Asia/Kolkata
View NTP servers       : chronyc sources -v
Force sync             : chronyc makestep
Restart chrony         : systemctl restart chronyd

APPLY & VERIFY
--------------
Apply changes          : nmcli connection up ens160
Restart NM             : systemctl restart NetworkManager
Check NM status        : systemctl status NetworkManager
Show connection        : nmcli connection show ens160
Show device            : nmcli device show ens160

SECTION 10: TROUBLESHOOTING
================================================================================

ISSUE                          FIX
-----------------------------  -------------------------------------------------
IP not showing after reboot    nmcli connection up ens160
DHCP IP keeps coming back      nmcli connection modify ens160 ipv4.method manual
No internet access             Check default gateway: ip r | grep default
DNS not resolving              Check: cat /etc/resolv.conf
NTP not syncing                systemctl restart chronyd && chronyc makestep
Route not working              Check gateway is reachable: ping 192.168.121.2
NM not managing interface      nmcli device set ens160 managed yes
Connection not activating      journalctl -u NetworkManager -f
